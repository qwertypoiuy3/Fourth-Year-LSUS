(define (rotate-once ls)
  (append (cdr ls) (list (car ls))))

(define (rotate-helper ls remaining)
  (if (null? remaining)
      '()
      (cons ls (rotate-helper (rotate-once ls) (cdr remaining)))))
  
  
(define (rotate ls) (rotate-helper ls ls))
